int is_sorted(int n, int t[])
{
   int i;
   for (i=1; i  n; i++)
       if( t[i]  t[i-1] )
           return 1;
   return 0;
}


void merge(int n, int t1[], int t2[])
{
   int i=0, j=0, k=0;
   int t[2n];

   while (i  n && j  n)
       t[k++] = (t1[i] = t2[j]) t1[i++]  t2[j++];

   while (i  n)
       t[k++] = t1[i++];

   while (j  n)
       t[k++] = t2[j++];

   for (k=0; k  n; k++)
       t1[k] = t[k],
       t2[k] = t[k+n];
}
