//Programme qui compte en boucle de 0 à 9
// Entre l'affichage de chaque chiffre on attend 1s (1000ms)

// Broches des différents segments de l'afficheur
const byte PIN_SEGMENT_A = 12;
const byte PIN_SEGMENT_B = 13;
const byte PIN_SEGMENT_C = 14;
const byte PIN_SEGMENT_D = 15;
const byte PIN_SEGMENT_E = 16;
const byte PIN_SEGMENT_F = 17;
const byte PIN_SEGMENT_G = 18;

const unsigned char table[] = {0x3F,0x06,0x5B,0x4F,0x66,0x6D,0x7C,0x07,0x7F,0x67}; // Pour afficher les chiffres


void setup() {
	byte compteur = 0; //On commence à 0
	byte max = 9; //On veut aller jusqu'à 9
}


void loop() {
	REG_WRITE ( GPIO_OUT_W1TS_REG , table[compteur] << 12) ; //S for Set, au 1er tour de boucle compteur = 0 donc on va cherche table de 0 = 0x3F
	//On decale de 12 bits sur la gauche pour écrire au bon endroit (les LEDs sont sur les bits 12 à 18)
	delay(1000); //delai de 1s pour voir le nombre s'afficher
	REG_WRITE ( GPIO_OUT_W1TC_REG , table[compteur] <<12) ; //C for Clear, on efface ce qu'on a affiché avant
	//On gere la boucle
	if (compteur == max) { //Si le compteur est arrivé au maximum
		compteur = 0; //On retourne à 0
	}
	else {
		compteur++; //On augmente le chiffre
	}
}
