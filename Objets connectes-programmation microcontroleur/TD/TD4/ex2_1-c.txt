char colonne1 = 13;
char colonne2 = 14;
char colonne3 = 15;

char compteur = 0;

void setup(){
	pinMode(colonne1,INPUT);
	pinMode(colonne2,INPUT);
	pinMode(colonne3,INPUT);
}

void loop(){
	if(digitalRead(colonne1 + compteur) == HIGH) {
		//le bouton est relaché sinon on fait rien
		Serial.print(colonne1 + compteur); // On affiche le numéro de la colonne
	}
	if (compteur >= 2){
		compteur = 0;
	}
	else {
		compteur++;
	}
}
