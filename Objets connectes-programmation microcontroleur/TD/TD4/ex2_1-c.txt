char colonne1 = 13;
char colonne2 = 14;
char colonne3 = 15;

void setup(){
	pinMode(colonne1,INPUT);
	pinMode(colonne2,INPUT);
	pinMode(colonne3,INPUT);
}

void loop(){
	if(digitalRead(colonne1) == HIGH) {
		//le bouton est relaché sinon on fait rien
		Serial.print(colonne1); // On affiche le numéro de la colonne
	}
	
	if(digitalRead(colonne2) == HIGH) {
		Serial.print(colonne2);
	}
	
	if(digitalRead(colonne3) == HIGH) {
		Serial.print(colonne3);
	}
}
