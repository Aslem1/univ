const char colonne1 = 13;
const char colonne2 = 14;
const char colonne3 = 15;

const char ligneA = 16;
const char ligneB = 17;
const char ligneC = 18;
const char ligneD = 19;

char compteurLignes = 0;
char compteurColonnes = 0;

void setup(){
	pinMode(colonne1,INPUT); // On lit, donc INPUT
	pinMode(colonne2,INPUT);
	pinMode(colonne3,INPUT);
	
	pinMode(ligneA,OUTPUT); // On écrit, donc OUTPUT
	pinMode(ligneB,OUTPUT);
	pinMode(ligneC,OUTPUT);
	pinMode(ligneD,OUTPUT);
}

void loop(){
	digitalWrite(ligneA + compteurLignes, HIGH);

	if(digitalRead(colonne1 + compteurColonnes) == HIGH) {
		Serial.print(colonne1 + compteurColonnes);
		Serial.print(ligneA + compteurLignes); //On affiche le numero de la ligne + le compteur
	}
	if (compteurColonnes >= 2){
		compteurColonnes = 0;
	}
	else {
		compteurColonnes++;
	}
	
	digitalWrite(ligneA + compteurLignes, LOW);
	
	if (compteurLignes >= 3) { // Si on arrive à la 4ème ligne (la dernière)
		compteurLignes = 0; // On remet le compteur à 0
	}
	else {
		compteurLignes++; //Sinon on continue d'incrémenter le compteur
	}
}
